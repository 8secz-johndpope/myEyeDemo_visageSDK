284  /* Get normalized face image.
285  *
286  * This function returns normalized face image with corresponding feature points.
287  * Size of the normalized face in the image is such that interpupillary distance is approximately quarter of the image width.
288  *
289  * Face will be normalized to a varying degree depending on normalization type. For example rotated
290  * face with open mouth will only have its pose straightened with normalization type VS_NORM_POSE while
291  * with addition of VS_NORM_AU normalized face will also have closed mouth.
292  *
293  * Note that the face will always have its pose straightened.
294  *
295  * Types of normalization are:
296  * - VS_NORM_POSE - face translation and rotation are set to zero thereby normalizing the pose
297  * - VS_NORM_SU - parameters describing the face shape (shape units) are set to zero thereby normalizing the face shape
298  * - VS_NORM_AU - parameters describing facial movements (action units) are set to zero, for example open mouth will be closed
299  *
300  * Different types of normalization can be combined with "|" operator, for example VS_NORM_POSE | VS_NORM_SU.
301  *
302  * \b Note: To use this functionality a FaceTrack or FaceDetect license is required.
303  *
304  * @param frame image containing the face to be normalized, must be grayscale
305  * @param normFace image containing the normalized face; it must be allocated before calling the function; face size will depend on this image size
306  * @param face_data FaceData structure containing the information about the face that will be normalized
307  * @param normFDP features points that correspond to the normalized face; coordinates are normalized to 0-1 range
308  * @param norm_type normalization type, a binary combination of VS_NORM_POSE - normalizes pose, VS_NORM_SU - normalizes shape units and VS_NORM_AU - normalizes action units
309  * @param dataPath path to the folder where Face Detector.cfg is located, default values is ""
310  */
311  void getNormalizedFaceImage (
312  VsImage* frame,
313  VsImage* normFace,
314  FaceData* face_data,
315  FDP* normFDP,
316  int norm_type = VS_NORM_POSE,
317  const char* dataPath = ""
318  );
319 
320 };
