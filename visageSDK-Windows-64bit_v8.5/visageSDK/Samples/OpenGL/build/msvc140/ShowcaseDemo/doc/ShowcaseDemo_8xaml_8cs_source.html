55  * - detecting facial features, face recognition and age, gender and emotion estimation on multiple faces in still images, based on the VisageCSWrapper.VisageFeaturesDetector detector, VisageCSWrapper.VisageFaceRecognition and VisageCSWrapper.VisageFaceAnalyser

495  /// <summary>
496  /// Helper function for converting image data from Bitmap class to a raw byte array.
497  /// </summary>
498  /// <param name="bitmap">Bitmap image.</param>
499  /// <param name="stride">Image stride.</param>
500  /// <returns></returns>
501  private static byte[] BitmapToByteArray(Bitmap bitmap, ref int stride)
502  {
503  System.Drawing.Imaging.PixelFormat pixelFormat = bitmap.PixelFormat;
504  BitmapData bmpdata = bitmap.LockBits(new System.Drawing.Rectangle(0, 0, bitmap.Width, bitmap.Height), ImageLockMode.ReadOnly, pixelFormat);
505  int numbytes = bmpdata.Stride * bitmap.Height;
506  //((n + (k-1)) / k) * k;
507  //modify height to be the value of the next nearest number divisble by 4
508  int numbytes_padded = bmpdata.Stride * ((bitmap.Height + 3) / 4 * 4);
509  byte[] bytedata = new byte[numbytes_padded];
510  stride = bmpdata.Stride;
511  IntPtr ptr = bmpdata.Scan0;
512 
513  Marshal.Copy(ptr, bytedata, 0, numbytes);
514 
515  bitmap.UnlockBits(bmpdata);
516 
517  return bytedata;
518  }

520  /// <summary>
521  /// Worker event for tracking/detection from an image source that runs in a separate thread. Used in runTrackFromImage() function.
522  ///
523  /// Expects two arguments sent via e parameter: path to the image and currently active VISAGE_MODE (TRACKER, DETECTOR). Based on the current mode initiates tracking or detection.
524  /// Depending on the tracker return status runs face analysis and face recognition in a separate thread.
525  ///
526  /// Locking mechanism is used to synchronize frame and tracking results rendering. Due to face analysis and face recognition running in separate thread
527  /// rendering of age, gender, emotion and identity is not synchronized with the rendering of the frame.
528  ///
529  /// Implemented using <see cref="BackgroundWorker"/>.
530  /// </summary>
531  /// <param name="sender"></param>
532  /// <param name="e"></param>
533  private void worker_DoWorkImage(object sender, DoWorkEventArgs e)
534  {
535  BackgroundWorker worker = sender as BackgroundWorker;
536  List<object> argumentList = e.Argument as List<object>;
537  string file = (string)argumentList[0];
538  VISAGE_MODE currentMode = (VISAGE_MODE)argumentList[1];
539 
540  //Age estimation enabled by default
541  Dispatcher.BeginInvoke((Action)(() => { enableAgeEstimCheckbox(); }));
542 
543  //Load image data by filename
544  Bitmap bmp = new Bitmap(file);
545  int bitmapStride = 0;
546  byte[] bmp_byte = BitmapToByteArray(bmp, ref bitmapStride);
547  BitmapData bmpdata = bmp.LockBits(new System.Drawing.Rectangle(0, 0, bmp.Width, bmp.Height), ImageLockMode.ReadOnly, bmp.PixelFormat);
548 
549  //Determine image format
550  int format = (int)VISAGE_FORMAT.BGR;
551  int origin = (int)VISAGE_ORIGIN.TL;
552  int nChannels = 3;
553  switch (bmp.PixelFormat)
554  {
555  case PixelFormat.Format24bppRgb:
556  format = (int)VISAGE_FORMAT.BGR;
557  nChannels = 3;
558  break;
559  case PixelFormat.Format8bppIndexed:
560  format = (int)VISAGE_FORMAT.LUMINANCE;
561  nChannels = 1;
562  Dispatcher.BeginInvoke((Action)(() => { disableAgeEstimCheckbox(); }));
563  break;
564  case PixelFormat.Format32bppRgb:
565  format = (int)VISAGE_FORMAT.BGRA;
566  nChannels = 4;
567  Dispatcher.BeginInvoke((Action)(() => { disableAgeEstimCheckbox(); }));
568  break;
569  default:
570  format = -1;
571  break;
572  }
573 
574  //Clear FA variables
575  lock (lockFaceData)
576  {
577  InitializeFaceAnalysis();
578  }
579 
580  //Create a VSImage and fill it with data to pass it to the DetectFeatures function
581  frame = new VSImage(bmp.Width, bmp.Height, VS_DEPTH._8U, nChannels);
582  frame.imageData = bmp_byte;
583 
584  byte[] bytedataRGB = new byte[bmp_byte.Length];
585 
586  convertFromBGRToRGB(ref bytedataRGB, ref bmp_byte);
587 
588  frameRGB = new VSImage(bmp.Width, bmp.Height, VS_DEPTH._8U, nChannels);
589  frameRGB.imageData = bytedataRGB;
590 
591  while (true)
592  {
593  //Check if the worker thread was canceled
594  if (worker.CancellationPending || format == -1)
595  {
596  isTracked = false;
597  e.Cancel = true;
598  break;
599  }
600  else if (currentMode == VISAGE_MODE.TRACKER)
601  {
602  //
603  FaceData[] analysisData = new FaceData[maxFacesTracker];
604 
605  //Lock the shared variables so there can be no read/write concurrency between the tracking thread and the rendering thread
606  lock (lockFaceData)
607  {
608  pureTrackStopwatch.Start();
609  startTime = pureTrackStopwatch.ElapsedMilliseconds;
610  gTrackerStatus = gVisageTracker.Track(bmp.Width, bmp.Height, bmp_byte, gFaceData, format, origin, bmpdata.Stride, -1, maxFacesTracker);
611  pureTrackStopwatch.Stop();
612  endTime = pureTrackStopwatch.ElapsedMilliseconds;
613  elapsedTime = endTime - startTime;
614 
615  //While locked, copy face data from the tracker to face data array for face analysis and recognition
616  for (int i = 0; i < maxFacesTracker; ++i)
617  {
618  analysisData[i] = new FaceData(gFaceData[i]);
619  }
620 
621  //Signalize to the rendering thread that the tracking is complete
622  isTracked = true;
623  }
624 
625  //Run face analysis on each face in a separate thread
626  runFaceAnalysis(frameRGB, analysisData, gTrackerStatus);
627 
628  //Run face recognition on each face in a separate thread
